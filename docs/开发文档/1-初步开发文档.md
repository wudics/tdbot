# 初步开发文档

> 基于 @docs/构思.md 的开发环境搭建与第一个 Electron + opencode 程序

## 目的

开发一个基于 opencode SDK 的桌面智能体，通过聊天窗口与用户交互，完成复杂任务。

## 架构

- **跨平台**: Linux / Windows / macOS
- **前端**: Electron（界面展示、用户交互、系统配置）
- **后端**: opencode SDK（智能体逻辑，内嵌 `opencode serve` 进程）
- **知识库**: WeKnora（单独模块，通过 API 管理）
- **智能体引擎**: opencode 二进制文件（随 Electron 打包）

## 技术选型

| 层面 | 方案 |
|------|------|
| UI 框架 | 纯 HTML / CSS / JS（无额外框架） |
| 进程通信 | Electron IPC（contextBridge + ipcRenderer/ipcMain） |
| opencode 启动 | 主进程 `createOpencode()` 启动内嵌服务器 |
| 流式输出 | SSE events → `delta` → IPC → DOM 直接追加 |
| Provider/Model | 硬编码 `deepseek/deepseek-v4-flash`（后续动态配置） |
| API Key | 直接写在代码中（后续改为动态配置） |

### 4. 主进程 — src/main/main.js

```javascript
const { app, BrowserWindow, ipcMain } = require("electron")
const path = require("path")

let mainWindow
let opencodeClient
let opencodeServer
let opencodeReady = false
const partTypes = {}

async function createWindow() {
  mainWindow = new BrowserWindow({
    width: 800,
    height: 600,
    webPreferences: {
      preload: path.join(__dirname, "preload.js"),
      contextIsolation: true,
      nodeIntegration: false,
    },
  })

  mainWindow.loadFile(path.join(__dirname, "..", "renderer", "index.html"))
}

app.whenReady().then(async () => {
  await createWindow()

  ipcMain.handle("start-chat", async () => {
    if (!opencodeReady) {
      return { success: false, error: "opencode 正在启动，请稍候重试" }
    }
    try {
      const events = await opencodeClient.event.subscribe()

      const session = await opencodeClient.session.create({
        title: "自我介绍",
      })

      await opencodeClient.session.promptAsync({
        sessionID: session.data.id,
        agent: "general",
        parts: [{ type: "text", text: "请介绍你自己" }],
      })

      const timeout = setTimeout(() => {
        mainWindow.webContents.send("stream-error", { message: "响应超时（60s）" })
      }, 60000)

      ;(async () => {
        try {
          for await (const event of events.stream) {
            if (event.type === "message.part.updated") {
              const part = event.properties.part
              if (part && part.id) {
                partTypes[part.id] = part.type
              }
            }
            if (event.type === "message.part.delta") {
              const delta = event.properties.delta
              const partID = event.properties.partID
              const partType = partTypes[partID] || "text"
              if (delta) {
                mainWindow.webContents.send("stream-chunk", { delta, partType })
              }
            }
            if (event.type === "session.idle") {
              clearTimeout(timeout)
              mainWindow.webContents.send("stream-end")
              break
            }
            if (event.type === "session.error") {
              clearTimeout(timeout)
              mainWindow.webContents.send("stream-error", event.properties.error)
              break
            }
          }
        } catch (streamErr) {
          clearTimeout(timeout)
          mainWindow.webContents.send("stream-error", { message: streamErr.message })
        }
      })()

      return { success: true }
    } catch (err) {
      console.error("start-chat error:", err)
      return { success: false, error: err.message }
    }
  })

  const { createOpencode } = await import("@opencode-ai/sdk/v2")
  const { client, server } = await createOpencode({
    config: {
      model: "deepseek/deepseek-v4-flash",
      agent: {
        general: {
          prompt: "你是一个名为 tdbot 的桌面智能体助手，由用户基于 opencode SDK 构建。请以 tdbot 的身份回答问题。",
        },
      },
      provider: {
        deepseek: {
          options: {
            apiKey: "sk-cd7b134f460b4fb297e16b75e9ffed3d",
          },
          models: {
            "deepseek-v4-flash": {
              id: "deepseek-v4-flash",
            },
          },
        },
      },
    },
  })
  opencodeClient = client
  opencodeServer = server
  opencodeReady = true

  app.on("before-quit", () => {
    if (opencodeServer) opencodeServer.close()
  })
})

app.on("window-all-closed", () => {
  if (opencodeServer) opencodeServer.close()
  if (process.platform !== "darwin") app.quit()
})
```

### 5. 预加载脚本 — src/main/preload.js

```javascript
const { contextBridge, ipcRenderer } = require("electron")

let cleanupStream = null
let cleanupEnd = null
let cleanupError = null

contextBridge.exposeInMainWorld("api", {
  startChat: () => ipcRenderer.invoke("start-chat"),
  onStream: (callback) => {
    if (cleanupStream) cleanupStream()
    const handler = (_event, data) => callback(data)
    ipcRenderer.on("stream-chunk", handler)
    cleanupStream = () => ipcRenderer.removeListener("stream-chunk", handler)
  },
  onEnd: (callback) => {
    if (cleanupEnd) cleanupEnd()
    const handler = () => callback()
    ipcRenderer.on("stream-end", handler)
    cleanupEnd = () => ipcRenderer.removeListener("stream-end", handler)
  },
  onError: (callback) => {
    if (cleanupError) cleanupError()
    const handler = (_event, error) => callback(error)
    ipcRenderer.on("stream-error", handler)
    cleanupError = () => ipcRenderer.removeListener("stream-error", handler)
  },
})
```

### 8. 渲染逻辑 — src/renderer/renderer.js

```javascript
const messagesEl = document.getElementById("messages")
const btnStart = document.getElementById("btn-start")

let textEl = null
let reasoningEl = null

btnStart.addEventListener("click", async () => {
  btnStart.disabled = true
  btnStart.textContent = "对话中..."

  const userMsg = document.createElement("div")
  userMsg.className = "message user"
  userMsg.textContent = "请介绍你自己"
  messagesEl.appendChild(userMsg)

  const assistantEl = document.createElement("div")
  assistantEl.className = "message assistant"
  messagesEl.appendChild(assistantEl)

  reasoningEl = document.createElement("div")
  reasoningEl.className = "reasoning"
  assistantEl.appendChild(reasoningEl)

  textEl = document.createElement("div")
  textEl.className = "text"
  assistantEl.appendChild(textEl)

  window.api.onStream((data) => {
    const el = data.partType === "reasoning" ? reasoningEl : textEl
    el.textContent += data.delta
    messagesEl.scrollTop = messagesEl.scrollHeight
  })

  window.api.onEnd(() => {
    if (reasoningEl && !reasoningEl.textContent.trim()) {
      reasoningEl.style.display = "none"
    }
    btnStart.disabled = false
    btnStart.textContent = "开始对话"
  })

  window.api.onError((error) => {
    textEl.textContent = `错误: ${JSON.stringify(error)}`
    btnStart.disabled = false
    btnStart.textContent = "开始对话"
  })

  const result = await window.api.startChat()
  if (!result.success) {
    textEl.textContent = `启动失败: ${result.error}`
    btnStart.disabled = false
    btnStart.textContent = "开始对话"
  }
})
```

## 运行

```bash
npm start
```

预期效果：
1. Electron 窗口打开
2. 点击"开始对话"按钮
3. 渲染进程通过 IPC 通知主进程
4. 主进程启动 opencode、订阅 events、创建 session、发送 prompt
5. 事件流中的 `message.part.updated` delta 通过 IPC 逐块发送到渲染进程
6. 渲染进程将 delta 直接追加到 DOM，实现自然流式输出
7. `session.idle` 事件触发后，按钮恢复可用状态

```
    │                   │ ─────────────────────────>│
    │                   │ client.auth.set()        │
    │                   │ ─────────────────────────>│
    │                   │ event.subscribe()        │
    │                   │ ─────────────────────────>│
    │                   │ session.create()         │
    │                   │ ─────────────────────────>│
    │                   │ session.promptAsync()    │
    │                   │ ─────────────────────────>│
    │                   │                          │
    │                   │ ◄── message.part.updated─│
    │ stream-chunk(delta)│     (delta: "\n\n")     │
    │ ◄─────────────────│                          │
    │ textContent += "\n\n"                        │
    │                                              │
    │                   │ ◄── message.part.updated─│
    │ stream-chunk(delta)│     (delta: "我是")     │
    │ ◄─────────────────│                          │
    │ textContent += "我是"                        │
    │                   │         ...              │
    │                   │ ◄── session.idle ────────│
    │ stream-end        │                          │
    │ ◄─────────────────│                          │
    │ 按钮恢复可用                                │
```
